enable
configure terminal
hostname 3R4
ip routing

! LACP hacia 3R2
interface Port-channel5
  no switchport
  ip address 172.39.0.86 255.255.255.252
  no shutdown
exit
interface range FastEthernet0/1-3
  no switchport
  no ip address
  channel-group 5 mode active
  no shutdown
exit

! VLAN 53
vlan 53

exit

! SVI para VLAN 53
interface Vlan53
  ip address 192.168.33.1 255.255.255.192
  ip helper-address 192.168.13.194
  no shutdown
exit

! Trunk hacia Switch 3S4
interface FastEthernet0/24
  switchport trunk encapsulation dot1q
  switchport mode trunk
  switchport trunk allowed vlan 53
  no shutdown
exit

! OSPF
router ospf 1
  network 172.39.0.84 0.0.0.3 area 0
  network 192.168.33.0 0.0.0.127 area 0
  passive-interface Vlan53
exit
